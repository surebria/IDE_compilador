// Ejercicio 3: Mayor de Tres Numeros Reales (Selectivo)
main {
    // Declaracion de variables
    float num1, num2, num3, mayor;

    // Entrada de datos: Lecturas individuales
    cin >> num1;
    cin >> num2;
    cin >> num3;

    // Logica para encontrar el mayor
    mayor = num1;

    if num2 > mayor then
        mayor = num2;
    end

    if num3 > mayor then
        mayor = num3;
    end
    
    // Salida de resultados
    cout << mayor;
}
